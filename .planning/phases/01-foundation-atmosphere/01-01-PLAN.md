    Configure `client/tailwind.config.ts` with FRACTAL's design tokens:

    ```typescript
    import type { Config } from 'tailwindcss'

    export default {
      content: ['./index.html', './src/**/*.{js,ts,jsx,tsx}'],
      theme: {
        extend: {
          colors: {
            background: {
              dark: '#0f172a',    // slate-900 - deepest background
              DEFAULT: '#1e293b', // slate-800 - standard background
              light: '#334155',   // slate-700 - elevated surfaces
            },
            accent: {
              warm: '#f59e0b',    // amber-500 - primary warm accent
              heat: '#f97316',    // orange-500 - intense heat/fire
              glow: '#fbbf24',    // amber-400 - soft cabin glow
            },
            frost: {
              DEFAULT: '#e2e8f0', // slate-200 - frost/ice color
              light: '#f1f5f9',   // slate-100 - snow white
              blue: '#bfdbfe',    // blue-200 - ice blue tint
            },
          },
          fontFamily: {
            heading: ['Fredoka', 'sans-serif'],
            body: ['Nunito', 'sans-serif'],
          },
          zIndex: {
            background: '0',
            canvas: '10',
            particles: '20',
            overlay: '30',
            hud: '40',
          },
        },
      },
      plugins: [],
    } satisfies Config
    ```

    Create `client/src/styles/globals.css` with:

    ```css
    @tailwind base;
    @tailwind components;
    @tailwind utilities;

    /* Z-index layer system */
    :root {
      --z-background: 0;
      --z-canvas: 10;
      --z-particles: 20;
      --z-overlay: 30;
      --z-hud: 40;
    }

    /* Base styles */
    html, body, #root {
      margin: 0;
      padding: 0;
      width: 100vw;
      height: 100vh;
      overflow: hidden;
      background-color: #0f172a;
      font-family: 'Nunito', sans-serif;
    }

    /* Heading typography */
    h1, h2, h3, h4, h5, h6 {
      font-family: 'Fredoka', sans-serif;
    }

    /* React Flow container must have explicit dimensions */
    .react-flow-wrapper {
      width: 100%;
      height: 100%;
    }
    ```

    Update `client/src/main.tsx` to import fonts and styles in correct order:

    ```typescript
    // Fonts first
    import '@fontsource/fredoka/400.css'
    import '@fontsource/fredoka/700.css'
    import '@fontsource/nunito/400.css'
    import '@fontsource/nunito/600.css'
    // Then global styles
    import './styles/globals.css'
    // Then React
    import { StrictMode } from 'react'
    import { createRoot } from 'react-dom/client'
    import App from './App'

    createRoot(document.getElementById('root')!).render(
      <StrictMode>
        <App />
      </StrictMode>,
    )
    ```

    Update `client/src/App.tsx` to verify fonts and colors work:
